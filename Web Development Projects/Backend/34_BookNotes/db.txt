CREATE TABLE IF NOT EXISTS public.books
(
    id integer NOT NULL DEFAULT nextval('books_id_seq'::regclass),
    isbn character varying(20) COLLATE pg_catalog."default",
    title character varying(50) COLLATE pg_catalog."default",
    author character varying(50) COLLATE pg_catalog."default",
    description character varying(300) COLLATE pg_catalog."default",
    review character varying(500) COLLATE pg_catalog."default",
    rating integer,
    created character varying(100) COLLATE pg_catalog."default",
    createddate timestamp without time zone,
    CONSTRAINT books_pkey PRIMARY KEY (id)
)
